@app.route("/view-library") def view_library(): genre = request.args.get("genre") connection_obj = sqlite3.connect('CS4750Project.db') cursor_obj = connection_obj.cursor() if genre: cursor_obj.execute(f"SELECT * FROM Game WHERE Genre = '{genre}'") else: cursor_obj.execute("SELECT * FROM Game") games = cursor_obj.fetchall() connection_obj.close() return render_template("view-library.html", games=games)
